/**
 * ============================================
 * THREE ROLES & PERMISSIONS
 * ============================================
 */

// ===== ADMIN =====
ADMIN CAN:
✓ Add/Update/Delete teachers
✓ Add/Update/Delete students
✓ View all classes, teachers, students
✓ Manage settings
✓ View all reports and data
✓ Manage academic years
✓ Access everything

ADMIN CANNOT:
✗ Nothing (full access)

FRONTEND: Shows "Admin Dashboard"
ROUTES: /api/teachers (all), /api/students (all), /api/settings

// ===== TEACHER =====
TEACHER CAN:
✓ View assigned classes
✓ View students in their classes
✓ Take attendance
✓ Create assignments and exams
✓ Grade students
✓ View class reports

TEACHER CANNOT:
✗ Add/Update/Delete other teachers
✗ Add/Update/Delete students
✗ Manage system settings
✗ Access other teacher's classes
✗ Delete exams/assignments

FRONTEND: Shows "Teacher Dashboard"
ROUTES: /api/teachers/classes, /api/students (own only), /api/attendance (own)

// ===== STUDENT =====
STUDENT CAN:
✓ View own dashboard
✓ See own attendance
✓ View own assignments
✓ View own exams and grades
✓ View own timetable
✓ Submit assignments

STUDENT CANNOT:
✗ Add/Manage any users
✗ Take attendance
✗ Create exams
✗ View other student's data
✗ Change settings
✗ Delete anything

FRONTEND: Shows "Student Dashboard"
ROUTES: /api/students/:ownId (self only), /api/attendance (own)

/**
 * ============================================
 * BACKEND ROUTE PATTERNS
 * ============================================
 */

// PATTERN 1: Admin Only
router.post('/admin/action', protect, authorize(['admin']), handler);
// Only admins can POST

// PATTERN 2: Admin OR Teacher
router.get('/classes', protect, authorize(['admin', 'teacher']), handler);
// Both admins and teachers can GET /classes
// But teachers see only their classes (checked in controller)

// PATTERN 3: Admin OR Teacher OR Student
router.get('/dashboard', protect, authorize(['admin', 'teacher', 'student']), handler);
// Anyone can see their own dashboard
// Students see own dashboard, teachers see class dashboard, admins see system dashboard

// PATTERN 4: With Permission Check
router.post('/exam', protect, checkPermission('createExam'), handler);
// Based on role-based permissions, not just role

// PATTERN 5: With Ownership Check
router.get('/profile/:id', protect, authorize(['admin', 'student']), checkOwnership, handler);
// Admin can see any profile
// Student can only see their own profile (checkOwnership validates this)

{
  "success": false,
  "message": "Not authorized, no token provided"
}

// 403 Forbidden (User doesn't have permission)
{
  "success": false,
  "message": "Access denied. Your role 'student' is not authorized for this action.",
  "userRole": "student",
  "requiredRoles": ["admin", "teacher"]
}

// 403 Forbidden (Specific permission denied)
{
  "success": false,
  "message": "Permission denied. You don't have 'createExam' permission.",
  "userRole": "student",
  "requiredPermission": "createExam"
}

 * ============================================
 */

// View Routes (Read-Only)
authorize(['admin', 'teacher', 'student']) // Everyone
authorize(['admin', 'teacher'])             // Admin & Teacher
authorize(['admin'])                        // Admin Only

// Create Routes
authorize(['admin', 'teacher'])             // Can create (exams, assignments)
authorize(['admin'])                        // Can create only (users, settings)

// Delete Routes
authorize(['admin'])                        // Admin Only (ALWAYS)

// Data Access Routes
authorize(['admin', 'student']) + checkOwnership  // View own data
authorize(['admin', 'teacher', 'student'])        // View class data

 * ============================================
 * SECURITY PRINCIPLES

✓ Log all access denials

✓ Validate permissions in controller (not just route)
✓ Use specific roles, not generic flags
✓ Protect all DELETE operations with adminOnly
✓ Return least sensitive error messages
